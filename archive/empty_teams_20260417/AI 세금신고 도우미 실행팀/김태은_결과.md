```python
    def __init__(self):
        self.naver_client = NaverClovaClient(

        )
        self.google_client = vision.ImageAnnotatorClient()

    def process_receipt(self, image_s3_url):
        """
        1차: Naver Clova (비용 ₩0.8/장)
        2차: 신뢰도 <90% 시 Google Vision (₩1.5/장)
        예상 성공률: 82% Naver만 / 18% Google 추가
        """
        # Step 1: S3에서 이미지 다운로드
        image_bytes = self._download_from_s3(image_s3_url)
        
        # Step 2: Naver OCR 시도
        naver_result = self.naver_client.recognize(image_bytes)
        
        if naver_result['confidence'] >= 0.90:
            return {
                'provider': 'naver',
                'cost': 0.0008,  # USD
                'data': self._parse_naver(naver_result)
            }
        
        # Step 3: Google Vision Fallback
        google_result = self.google_client.document_text_detection(
            image={'content': image_bytes}
        )
        
        return {
            'provider': 'google',
            'cost': 0.0015,
            'data': self._parse_google(google_result)
        }
    
    def _parse_naver(self, raw_data):
        """Naver 응답→표준 스키마 변환"""
        return {
            'amount': self._extract_amount(raw_data['text']),
            'date': self._extract_date(raw_data['text']),
            'merchant': raw_data['business_name'],
            'category': self._classify_category(raw_data['text']),
            'confidence': raw_data['confidence']
        }
```
