#### 1.1 필수 테이블 DDL
```sql
-- users 테이블 (Supabase Auth 자동 생성, 확장만 추가)
create table public.profiles (
  id uuid references auth.users primary key,
  email text unique not null,
  created_at timestamptz default now(),
  subscription_tier text default 'free' check (subscription_tier in ('free', 'pro', 'team'))
);

-- projects 테이블
create table public.projects (
  id uuid default gen_random_uuid() primary key,
  user_id uuid references public.profiles not null,
  name text not null,
  api_key text unique not null, -- 'sk_live_' || encode(gen_random_bytes(32), 'hex')
  monthly_budget numeric(10,2) default 100.00,
  current_spend numeric(10,2) default 0.00,
  created_at timestamptz default now()
);

-- token_usage 테이블 (실시간 집계용)
create table public.token_usage (
  id uuid default gen_random_uuid() primary key,
  project_id uuid references public.projects not null,
  model text not null, -- 'gpt-4', 'claude-3-opus' 등
  input_tokens int not null,
  output_tokens int not null,
  cost_usd numeric(10,6) not null, -- 모델별 단가 테이블 참조 계산
  created_at timestamptz default now()
);

-- 인덱스 (고빈도 쿼리 최적화)
create index idx_projects_user_id on projects(user_id);
create index idx_token_usage_project_created on token_usage(project_id, created_at desc);
```

#### 1.2 RLS 정책 (멀티테넌트 격리)
```sql
-- profiles: 본인만 읽기/쓰기
alter table profiles enable row level security;
create policy "Users can view own profile" on profiles
  for select using (auth.uid() = id);
create policy "Users can update own profile" on profiles
  for update using (auth.uid() = id);

-- projects: 본인 프로젝트만 접근
alter table projects enable row level security;
create policy "Users CRUD own projects" on projects
  using (user_id = auth.uid()); -- select/insert/update/delete 통합

-- token_usage: 본인 프로젝트만 조회
alter table token_usage enable row level security;
create policy "Users view own project usage" on token_usage
  for select using (
    project_id in (
      select id from projects where user_id = auth.uid()
    )
  );
```

**After (실제 연동)**:

```typescript
// app/api/auth/signup/route.ts
import { supabase } from '@/lib/supabase'

export async function POST(req: Request) {
  const { email, password } = await req.json()
  
  // 1. Supabase Auth 회원가입
  const { data: authData, error: authError } = await supabase.auth.signUp({
    email,
    password,
  })
  
  if (authError) {
    return NextResponse.json({ error: authError.message }, { status: 400 })
  }
  
  // 2. profiles 테이블에 초기 데이터 삽입
  const { error: profileError } = await supabase
    .from('profiles')
    .insert({
      id: authData.user!.id,
      email,
      subscription_tier: 'free'
    })
  
  if (profileError) {
    return NextResponse.json({ error: 'Profile creation failed' }, { status: 500 })
  }
  
  return NextResponse.json({ 
    user: authData.user,
    session: authData.session 
  })
}
```

### ⏱️ Phase 3: 대시보드 데이터 연동 (2시간)

#### 3.1 프로젝트 목록 조회 API
```typescript
// app/api/projects/route.ts
import { supabase } from '@/lib/supabase'

export async function GET(req: Request) {
  // 1. 현재 로그인 사용자 확인
  const { data: { session } } = await supabase.auth.getSession()
  if (!session) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
  }
  
  // 2. RLS 정책 자동 적용 (user_id = auth.uid() 필터링)
  const { data: projects, error } = await supabase
    .from('projects')
    .select('*')
    .order('created_at', { ascending: false })
  
  if (error) {
    return NextResponse.json({ error: error.message }, { status: 500 })
  }
  
  return NextResponse.json({ projects })
}
```

#### 3.2 프로젝트 생성 API
```typescript
// app/api/projects/route.ts
export async function POST(req: Request) {
  const { name, monthly_budget } = await req.json()
  
  const { data: { session } } = await supabase.auth.getSession()
  if (!session) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
  }
  
  // API 키 생성 (secure random)
  const apiKey = `sk_live_${crypto.randomBytes(32).toString('hex')}`
  
  const { data: project, error } = await supabase
    .from('projects')
    .insert({
      user_id: session.user.id,
      name,
      api_key: apiKey,
      monthly_budget: parseFloat(monthly_budget) || 100.00
    })
    .select()
    .single()
  
  if (error) {
    return NextResponse.json({ error: error.message }, { status: 500 })
  }
  
  return NextResponse.json({ project })
}
```
