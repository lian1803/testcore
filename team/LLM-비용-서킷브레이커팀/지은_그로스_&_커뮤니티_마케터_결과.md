### B. API 클라이언트 함수 (실제 fetch/axios 호출)
```typescript
// src/lib/api-client.ts
import axios from 'axios';
import { API_BASE_URL } from './api-config';

// Axios 인스턴스 생성 (토큰 자동 첨부)
const apiClient = axios.create({
  baseURL: API_BASE_URL,
  headers: {
    'Content-Type': 'application/json',
  },
});

// 요청 인터셉터 (JWT 토큰 자동 첨부)
apiClient.interceptors.request.use(
  (config) => {
    const token = localStorage.getItem('auth_token');
    if (token) {
      config.headers.Authorization = `Bearer ${token}`;
    }
    return config;
  },
  (error) => Promise.reject(error)
);

// 응답 인터셉터 (401 시 로그아웃)
apiClient.interceptors.response.use(
  (response) => response,
  (error) => {
    if (error.response?.status === 401) {
      localStorage.removeItem('auth_token');
      window.location.href = '/login';
    }
    return Promise.reject(error);
  }
);

// 회원가입
export async function signupUser(email: string, password: string) {
  try {
    const response = await apiClient.post('/auth/signup', { email, password });
    return response.data; // { userId, token }
  } catch (error: any) {
    throw new Error(error.response?.data?.message || '회원가입 실패');
  }
}

// 로그인
export async function loginUser(email: string, password: string) {
  const response = await apiClient.post('/auth/login', { email, password });
  const { token } = response.data;
  localStorage.setItem('auth_token', token);
  return response.data;
}

// 프로젝트 조회
export async function getProjects() {
  const response = await apiClient.get('/projects');
  return response.data; // [{ id, name, budget, tokenUsed }]
}

// 프로젝트 생성
export async function createProject(name: string, budget: number, llmProvider: string) {
  const response = await apiClient.post('/projects', {
    name,
    budget,
    llm_provider: llmProvider,
  });
  return response.data;
}

// 대시보드 데이터
export async function getDashboardData() {
  const response = await apiClient.get('/dashboard');
  return response.data; // { totalCost, tokenUsage, alerts }
}
```

### C. React Hook으로 API 호출 (UI 컴포넌트에서 사용)
```typescript
// src/hooks/useProjects.ts
import { useState, useEffect } from 'react';
import { getProjects, createProject } from '@/lib/api-client';

export function useProjects() {
  const [projects, setProjects] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    async function fetchProjects() {
      try {
        const data = await getProjects();
        setProjects(data);
      } catch (err: any) {
        setError(err.message);
      } finally {
        setLoading(false);
      }
    }
    fetchProjects();
  }, []);

  const addProject = async (name: string, budget: number, provider: string) => {
    try {
      const newProject = await createProject(name, budget, provider);
      setProjects([...projects, newProject]);
      return newProject;
    } catch (err: any) {
      throw new Error(err.message);
    }
  };

  return { projects, loading, error, addProject };
}
```

### D. UI 컴포넌트에서 실제 데이터 사용
```tsx
// src/app/dashboard/page.tsx
'use client';
import { useProjects } from '@/hooks/useProjects';

export default function DashboardPage() {
  const { projects, loading, error, addProject } = useProjects();

  if (loading) return <div>로딩 중...</div>;
  if (error) return <div className="text-red-500">오류: {error}</div>;

  return (
    <div>
      <h1>프로젝트 목록</h1>
      {projects.map((p: any) => (
        <div key={p.id}>
          {p.name} - 예산: ${p.budget}
        </div>
      ))}
      <button onClick={() => addProject('새 프로젝트', 1000, 'openai')}>
        프로젝트 추가
      </button>
    </div>
  );
}
```

### 필수 테스트 플로우 (3회 반복)
1. **회원가입**
   - 이메일/비밀번호 입력 → 성공 메시지 확인
   - 중복 이메일 → 에러 메시지 "이미 존재하는 이메일입니다"
   
2. **로그인**
   - 올바른 계정 → 대시보드 이동
   - 잘못된 비밀번호 → "이메일 또는 비밀번호가 틀렸습니다"

3. **프로젝트 생성**
   - 이름/예산/LLM 선택 → 목록에 즉시 표시
   - 예산 0원 입력 → "예산은 최소 $1 이상이어야 합니다"

4. **대시보드 조회**
